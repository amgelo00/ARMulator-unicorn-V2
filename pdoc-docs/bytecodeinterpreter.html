 62    def setBreakpointInstr(self, listLineNumbers):
 63        """
 64        Set breakpoints at the line numbers specified.
 65
 66        :param listLineNumbers: an iterable
 67        """
 68        # First, we remove all execution breakpoints
 69        self.sim.mem.removeExecuteBreakpoints(removeList=[self.line2addr[b] for b in self.lineBreakpoints])
 70
 71        # Now we add all breakpoint
 72        # The easy case is when the line is directly mapped to a memory address (e.g. it is an instruction)
 73        # When it's not, we have to find the closest next line which is mapped
 74        # If there is no such line (we are asked to put a breakpoint after the last line of code) then no breakpoint is set
 75        self.lineBreakpoints = []
 76        for lineno in listLineNumbers:
 77            if lineno in self.line2addr:
 78                self.sim.mem.setBreakpoint(self.line2addr[lineno], 1)
 79                nextLine = lineno + 1
 80                while nextLine in self.line2addr and self.line2addr[nextLine] == self.line2addr[lineno]:
 81                    nextLine += 1
 82                if nextLine-1 not in self.lineBreakpoints:
 83                    self.lineBreakpoints.append(nextLine-1)

111    def toggleBreakpointMem(self, addr, mode):
112        """
113        Toggle a breakpoint in memory.
114
115        :param addr: Address to toggle the breakpoint
116        :param mode: breakpoint mode (see setBreakpointMem)
117        """
118        # Mode = 'r' | 'w' | 'rw' | 'e' | '' (passing an empty string removes the breakpoint)
119        modeOctal = 4*('r' in mode) + 2*('w' in mode) + 1*('e' in mode)
120        bkptInfo = self.sim.mem.toggleBreakpoint(addr, modeOctal)
121        if 'e' in mode and addr < self.bc['__MEMINFOEND']['CODE']:
122            addrprod4 = (addr // 4) * 4
123            if addrprod4 in self.addr2line:
124                if bkptInfo & 1:        # Add line breakpoint
125                    self.lineBreakpoints.append(self.addr2line[addrprod4][-1])
126                else:                   # Remove line breakpoint
127                    try:
128                        self.lineBreakpoints.remove(self.addr2line[addrprod4][-1])
129                    except ValueError:
130                        # Not sure how we can reach this, but just in case, it is not a huge problem so we do not want to crash
131                        pass

160        Set an interrupt.

215        """

237        Run the simulator in a given mode for one step only. Useful to execute step by step.

298        Set the value of a given address in memory.

302        :param val: the value to write, as a bytearray of one element.

306        if self.sim.mem._getRelativeAddr(addr, 1) is None:

312    def getCurrentInfos(self):
313        """
314        Return the current simulator state, with information relevant to the UI. The value is returned as :
315        [["highlightRead", ["r3", "SVC_r12", "z", "sz"]], ["highlightWrite", ["r1", "MEM_adresseHexa"]], ["nextline", 42], ["disassembly", ""]]

322        # Convert nextline from addr to line number
323        idx = [i for i, x in enumerate(s) if x[0] == "nextline"]

325            s[idx[0]][1] = self.addr2line[s[idx[0]][1]][-1]
